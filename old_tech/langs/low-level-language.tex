\subsection{Low-Level Language}

The next language is ,,low-level'', because it extends the previous one with speculation.

\myfig{
	\vspace{-1em}
  \begin{gather*}
  \begin{aligned}
  \mi{Final\ Result}\ \ird{\finalexprnoerr} \bnfdef&\ \ird{\valueexpr^{\sigma}} \mid \ird{x^{\sigma}} \hspace{0.5cm}
  \mi{May\ be\ a\ Result}\ \ird{\finalexpr} \bnfdef\ \ird{\finalexprnoerr} \mid \ird{stuck} \\
  %
  \mi{Expressions}~\ird{e} \bnfdef&\ \ird{\finalexpr} \mid \ird{e_{1} \oplus e_{2}} \mid \ird{x[e]} \mid \ird{getDIT\ x\ in\ e} \mid \ird{setDIT\ e} \\
  & \mid \ird{let\ x = e_{1}\ in\ e_{2}} \mid \ird{x[e_{1}]\leftarrow e_{2}} \\
  & \mid \ird{let\ x = new\ e_{1}\ in\ e_{2}} \mid \ird{delete\ x} \\
  & \mid \ird{return\ e} \mid \ird{call\ foo\ e} \mid \ird{ifz\ e_{1}\ then\ e_{2}\ else\ e_{3}} \\
  & \mid \ird{abort()} \mid \ird{x\ is\ \poisoned} \\
  & \mid \ird{\langle e_{1},e_{2}\rangle} \mid \ird{\pi_{1} e} \mid \ird{\pi_{2} e} \mid \ird{e\ has\ \type} \\
   &\text{where } \ird{\oplus}\in\{\ird{+},\ird{-},\ird{\times},\ird{<},\ird{/}\}\\
  %
  \mi{Functions}~\ird{\asymbol} \bnfdef&\ \ird{let\ foo\ x:= e} \hspace{0.25cm}
  \mi{Types}~\ird{\type} \bnfdef\ \ird{\nat} \mid \ird{\nat\times\nat} \\
  %
  \mi{Values}~\ird{\valueexpr} \bnfdef&\ \ird{n}\in\ird{\nat} \hspace{0.25cm}
  %
  \mi{References}~\ird{\loc}\in\ird{\nat} \\
  \mi{Eval. Ctx.}~\ird{K}\bnfdef &\ \ird{\hole{\cdot}}\mid \ird{K\oplus e}\mid \ird{v\oplus K}\mid \ird{x[K]}\mid \ird{let\ x=K\ in\ e}\\
                     &\ \mid\ird{x[K]\leftarrow e}\mid \ird{x[v]\leftarrow K}\mid \ird{let\ x=new\ K\ in\ e} \\
                     &\ \mid\ird{ifz\ K\ then\ e_1\ else\ e_2} \mid \ird{call\ foo\ K}\mid \ird{return\ K} \\
                     &\ \mid\ird{\langle K, e\rangle} \mid \ird{\langle \valueexpr, K \rangle } \mid \ird{\pi_{1} K} \mid \ird{\pi_{2} K} \mid \ird{K\ has\ \type} \mid \ird{barrier} \\
                     &\ \mid\ird{ifz\ K\ then\ e_{1}\ else\ e_{2}} \mid \ird{setDIT\ K} \\
  %
  \mi{Variables}~\ &\ird{x} \mid \ird{y} \mid \ird{foo} \mid \dots \hspace{0.5cm}
  %
  \mi{Poison}~\ \ird{\poison}~\bnfdef\ \ird{\poisonless} \mid \ird{\poisoned}\\
  \mi{Sandbox\ Tag}~&\ \ird{\sandboxtag}~\bnfdef\ \ird{\ctx} \mid \ird{\comp}\\
  %
  \mi{Typing. Env.}~\ird{\Gamma} \bnfdef&\ \ird{\hole{\cdot}} \mid \ird{\Gamma,x:\type}\hspace{0.5cm}
  \mi{Store}~\ird{\Delta} \bnfdef\ \ird{\hole{\cdot}} \mid \ird{x\mapsto(\loc;\sandboxtag;\poison;n),\Delta}\\
  %
  \mi{Communication}~\ird{\comm} \bnfdef&\ \ird{\ctxtocomp} \mid \ird{\comptoctx} \mid \ird{\nocomm} \hspace{0.5cm}
  \mi{Heaps}~\ird{H} \bnfdef\ \ird{\hole{\cdot}} \mid \ird{H}::\ird{n} \\
  %
  \mi{Cont.\ Stack}~\ird{\kontstack} \bnfdef&\ \ird{\hole{\cdot}} \mid (\ird{K};\ird{foo}),\ird{\kontstack} \hspace{0.5cm}
  \mi{Library}~\ird{\library} \bnfdef\ \ird{\hole{\cdot}} \mid \ird{\asymbol},\ird{\library} \\
  \mi{Relevant}\ \ird{\commlib} \bnfdef& \ird{\hole{\cdot}} \mid \ird{foo},\ird{\commlib}\hspace{0.5cm}
  \mi{State}~\ird{\Omega}\ \bnfdef\ \ird{\cfstate;\sandboxtag;n;\memstate}\\
  \mi{Flow\ State}~\ird{\cfstate}\bnfdef&\ \ird{\commlib;\library;\kontstack}\hspace{0.5cm}
  \mi{Memory\ State}~\ird{\memstate}\bnfdef\ \ird{H^{\ctx};H^{\comp};\Delta} \\
  %
  \mi{Programs}\ &\ird{prog\ \library_{ctx}\ \library_{comp}} \hspace{0.35cm}
  \mi{Substitutions}~\ird{\gamma} \bnfdef \ird{[\valueexpr/x],\gamma}\ \vert\ \ird{\hole{\cdot}} \\
  \mi{Security\ Tag}~\ird{\sigma} \bnfdef&\ \ird{\lock_{vX}} \mid \ird{\unlock} \\
  \mi{Leak\ Tag}~\ird{vX} \bnfdef&\ \ird{NONE} \mid \ird{PHT} 
  \end{aligned}
  \end{gather*}
}{mmlao-syntax}{Syntax of $\mmlAspec$}

The only difference between $\mmlAtscct$ and $\mmlAspec$ is speculative semantics.
