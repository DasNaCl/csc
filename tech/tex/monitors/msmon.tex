\subsubsection{\gls*{ms} Monitor}\label{subsubsec-mon-ms}
Instantiate \Cref{subsubsec-mon-dual} with \Cref{subsubsec-mon-tms,subsubsec-mon-sms}. 
We refer to the resulting monitor state as $\msMonitor$.
What is left to give is a translation from traces $\msTrace$ to traces $\varTrace[\tmssafe;\smssafe]$.
This translation is done as follows:
\begin{definition}{\definitionlabel[\gls*{ms} to \gls*{tms} and \gls*{sms}]{traces-ms-to-tmssms}}
  \begin{center}
    \judgbox{\msTrace\xlangrel{\mssafe}{\lrpars{\tmssafe;\smssafe}}\varTrace[\tmssafe;\smssafe]}{,,Project \gls*{ms}-traces to dual-trace \gls*{tms}-\gls*{sms}.''}

    \typerule{ms-to-tmssms-empty}{
    }{
      \hole{\cdot}\xlangrel{\mssafe}{\lrpars{\tmssafe;\smssafe}}\hole{\cdot}
    }{ms-to-tmssms-empty}
    %
    \typerule{ms-to-tmssms-cons}{
      \msEvent\xlangrel{\mssafe}{\tmssafe}\tmsEvent \rulesep
      \msEvent\xlangrel{\mssafe}{\smssafe}\smsEvent \rulesep
      \msTrace\xlangrel{\mssafe}{\lrpars{\tmssafe;\smssafe}}\varTrace[\tmssafe;\smssafe]
    }{
      \msEvent\cdot\msTrace\xlangrel{\mssafe}{\lrpars{\tmssafe;\smssafe}}\lrpars{\tmsEvent;\smsEvent}\cdot\varTrace[\tmssafe;\smssafe]
    }{ms-to-tmssms-cons}
  \end{center}
\end{definition}

\begin{definition}{\definitionlabel[\gls*{ms}-Satisfaction via Monitor]{trace-is-ms-via-mon}}
  \begin{align*}
    \mssafe\vdash\msTrace &\isdef \exists\varTrace[\tmssafe;\smssafe]\;\noncrashMonitor{\msMonitor},\monitorcheck{\emptyset}{\msMonitor}{\varTrace[\tmssafe;\smssafe]}\text{ and }
    \msTrace\xlangrel{\mssafe}{\lrpars{\tmssafe;\smssafe}}\varTrace[\tmssafe;\smssafe]
  \end{align*}
\end{definition}
