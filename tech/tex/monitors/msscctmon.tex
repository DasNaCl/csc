\subsubsection{\gls*{msscct} Monitor}\label{subsubsec-mon-msscct}

Instantiate \Cref{subsubsec-mon-dual} with \Cref{subsubsec-mon-ms,subsubsec-mon-scct}. 
We refer to the resulting monitor state as $\msscctMonitor$.
What is left to give is a translation from traces $\msTrace$ to traces $\varTrace[\mssafe;\scctsafe]$.
This translation is done as follows:
\begin{definition}{\definitionlabel[\gls*{ms} to \gls*{ms} and \gls*{scct}]{traces-ms-to-msscct}}
  \begin{center}
    \judgbox{\msTrace\xlangrel{\mssafe}{\lrpars{\mssafe;\scctsafe}}\varTrace[\mssafe;\scctsafe]}{,,Project \gls*{ms}-traces to dual-trace \gls*{ms}-\gls*{scct}.''}

    \typerule{ms-to-msscct-empty}{
    }{
      \hole{\cdot}\xlangrel{\mssafe}{\lrpars{\mssafe;\scctsafe}}\hole{\cdot}
    }{ms-to-mscct-empty}
    %
    \typerule{ms-to-msscct-cons}{
      \msEvent\xlangrel{\mssafe}{\scctsafe}\scctEvent \rulesep
      \msTrace\xlangrel{\mssafe}{\lrpars{\mssafe;\scctsafe}}\varTrace[\mssafe;\scctsafe]
    }{
      \msEvent\cdot\msTrace\xlangrel{\mssafe}{\lrpars{\mssafe;\scctsafe}}\lrpars{\msEvent;\scctEvent}\cdot\varTrace[\mssafe;\scctsafe]
    }{ms-to-msscct-cons}
  \end{center}
\end{definition}

\begin{definition}{\definitionlabel[\gls*{msscct}-Satisfaction via Monitor]{trace-is-msscct-via-mon}}
  \begin{align*}
    \msscctsafe\vdash\msTrace &\isdef \exists\varTrace[\mssafe;\scctsafe]\;\noncrashMonitor{\msscctMonitor},\monitorcheck{\emptyset}{\msscctMonitor}{\varTrace[\mssafe;\scctsafe]}\text{ and }
    \msTrace\xlangrel{\mssafe}{\lrpars{\mssafe;\scctsafe}}\varTrace[\mssafe;\scctsafe]
  \end{align*}
\end{definition}
